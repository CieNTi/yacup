47 {
48  /* Configure _dbg() */
49  #define YCP_FNAME "test_fsm_driver_simple"
50 
51  /* Testbench vars */
52  #define MAX_CYCLES 20
53  uint32_t current_cycle = MAX_CYCLES;
54 
55  /* FSM data and instance */
56  struct fsm_driver_simple_data fsm_driver_simple0_data;
57  struct fsm fsm_driver_simple0 =
58  {
59  .name = "fsm_driver_simple0",
60  .data = &fsm_driver_simple0_data
61  };
62 
63  /* Initialize fsm found at `fsm_driver_simple.c` */
64  _dbg("Initializing the finite-state machine\n");
65  if (fsm_init(&fsm_driver_simple0, fsm_driver_simple))
66  {
67  _dbg("- Cannot initialize the FSM. ERROR\n");
68  return 1;
69  }
70  _dbg("- Ok\n");
71 
72  /* Basics: enable + auto-restart + start as soon as possible */
73  fsm_enable(&fsm_driver_simple0);
74  fsm_set_loop(&fsm_driver_simple0);
75  fsm_request_start(&fsm_driver_simple0);
76 
77  /* fsm cycles */
78  fsm_print_info(&fsm_driver_simple0);
79  while (current_cycle--)
80  {
81  /* Start cycle */
82  _dbg("Cycle #%02u\n", MAX_CYCLES - current_cycle);
83 
84  /* Test the fsm */
85  if (fsm_do_cycle(&fsm_driver_simple0))
86  {
87  _dbg("Error when executing a fsm cycle\n");
88  fsm_print_info(&fsm_driver_simple0);
89  fsm_print_stats(&fsm_driver_simple0);
90  return 1;
91  }
92 
93  /* Finish cycle */
94  /* Simulate some time spent on the cycle */
95  nanosleep(&(struct timespec){ .tv_sec = 0, .tv_nsec = 100*1000000 }, NULL);
96  }
97 
98  /* Show some love */
99  fsm_print_stats(&fsm_driver_simple0);
100 
101  /* Cya! */
102  _dbg("If you are reading this, everything went correctly :_)\n");
103  return 0;
104 
105  /* Free _dbg() config */

108 }
