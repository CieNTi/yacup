31 /* Print `fsm` information to STDOUT.
32  * Read `yacup/fsm/debug.h` for complete information. */
33 void fsm_print_info(struct fsm *fsm)
34 {
35  /* Configure _dbg() */
36  #define YCP_FNAME "fsm_print_info"
37 
38  if (fsm == NULL) { return; }
39  _dbg("name .......: %s\n", fsm->name);
40  _dbg("config .....: %u\n", fsm->config);
41  _dbg("state ......: %u\n", fsm->state);
42  _dbg("driver .....: %p\n", (void *)(size_t)fsm->driver);
43  _dbg(" .start ....: %p\n", (void *)(size_t)fsm->driver->start);
44  _dbg(" .stop .....: %p\n", (void *)(size_t)fsm->driver->stop);
45  _dbg("last .......: %p\n", (void *)(size_t)fsm->last);
46  _dbg("now ........: %p\n", (void *)(size_t)fsm->now);
47  _dbg("next .......: %p\n", (void *)(size_t)fsm->next);
48 
49  /* Free _dbg() config */
50  #undef YCP_FNAME
51 }

53 /* Print `fsm` stats to STDOUT.
54  * Read `yacup/fsm/debug.h` for complete information. */
55 void fsm_print_stats(struct fsm *fsm)
56 {
57  /* Configure _dbg() */
58  #define YCP_FNAME "fsm_print_stats"
59 
60  if (fsm == NULL) { return; }
61  _dbg("FSM_ALL ...: %lu\n", fsm->stats[FSM_ALL]);
62  _dbg("FSM_NEW ...: %lu\n", fsm->stats[FSM_NEW]);
63  _dbg("FSM_PAUSE .: %lu\n", fsm->stats[FSM_PAUSE]);
64  _dbg("FSM_RUN ...: %lu\n", fsm->stats[FSM_RUN]);
65  _dbg("FSM_ERROR .: %lu\n", fsm->stats[FSM_ERROR]);
66 
67  /* Free _dbg() config */
68  #undef YCP_FNAME
69 }
