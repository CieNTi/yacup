27 /* Initializes a `ce_codec`
28  * Read `yacup/ce/codec.h` for complete information. */
29 int ce_codec_init(struct ce_codec *codec,
30  int (*ce_codec_driver_init)(struct ce_codec *))
31 {
32  /* Configure _dbg() */
33  #define YCP_FNAME "ce_codec_init"
34 
35  if (/* Invalid codec? */
36  (codec == NULL) ||
37  /* Invalid driver init? */
38  (ce_codec_driver_init == NULL))
39  {
40  _dbg("Invalid codec or driver init function\n");
41  return 1;
42  }
43 
44  /* Presets */
45  codec->encode.command = NULL;
46  codec->encode.data = NULL;
47  codec->encode.message = NULL;
48  codec->decode.command = NULL;
49  codec->decode.data = NULL;
50  codec->decode.message = NULL;
51 
52  /* Now call the driver init function, and go */
53  if (ce_codec_driver_init(codec))
54  {
55  _dbg("Failed channel initialization\n");
56  return 1;
57  }
58 
59  /* Assign default name, if not previously set */
60  if (codec->name == NULL)
61  {
62  codec->name = YCP_NAME;
63  }
64 
65  /* Ok! */
66  _dbg("Codec '%s' initialized successfully\n", codec->name);
67  return 0;
68 
69  /* Free _dbg() config */
70  #undef YCP_FNAME
71 }
