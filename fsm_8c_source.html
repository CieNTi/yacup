32 /* Initializes a `fsm` common data, then lower level init, passed by argument.
33  * Read `yacup/fsm.h` for complete information. */
34 int fsm_init(struct fsm *fsm, int (*fsm_driver_init)(struct fsm *))
35 {
36  /* Configure _dbg() */
37  #define YCP_FNAME "fsm_init"
38 
39  if (/* Invalid fsm? */
40  (fsm == NULL) ||
41  /* Invalid driver init? */
42  (fsm_driver_init == NULL))
43  {
44  _dbg("Invalid fsm or driver init function\n");
45  return 1;
46  }
47 
48  /* Fill FSM common data */
49  fsm->config = 0;
50  fsm->state = FSM_NEW;
51  fsm->stats[FSM_NEW] = 0;
52  fsm->stats[FSM_PAUSE] = 0;
53  fsm->stats[FSM_RUN] = 0;
54  fsm->stats[FSM_ERROR] = 0;
55  fsm->stats[FSM_ALL] = 0;
56 
57  /* Now call the driver init function, and go */
58  if (fsm_driver_init(fsm))
59  {
60  _dbg("Failed fsm initialization\n");
61  return 1;
62  }
63 
64  /* Assign default name, if not previously set */
65  if (fsm->name == NULL)
66  {
67  fsm->name = YCP_NAME;
68  }
69 
70  /* Ok! */
71  _dbg("fsm '%s' initialized successfully\n", fsm->name);
72  return 0;
73 
74  /* Free _dbg() config */
75  #undef YCP_FNAME
76 }

126 /* Stepper function (single cycle and exit, need outside controller loop)
127  * Read `yacup/fsm.h` for complete information. */
128 int fsm_do_cycle(struct fsm *fsm)
129 {
130  /* Configure _dbg() */
131  #define YCP_FNAME "fsm_do_cycle"
132 
133  if (fsm == NULL)
134  {
135  _dbg("fsm is NULL. ERROR\n");
136 
137  /* Totally wrong, return dramatically */
138  return 1;
139  }
140 
141  if (!(fsm->config & FSM_CONFIG_ENABLED))
142  {
143  _dbg("%s@disabled\n", fsm->name);
144 
145  /* Disabled, return silently */
146  return 0;
147  }
148 
149  /* Record FSM_ALL stat */
150  fsm->stats[FSM_ALL]++;
151 
152  /* Behavior */
153  switch (fsm->state)
154  {
155  /* This should be a 1-pass state only, only initialization and checks */
156  case FSM_NEW:
157  _dbg("%s@FSM_NEW\n", fsm->name);
158  _dbg("fsm has been created. Lets configure it\n");
159 
160  /* Record FSM_NEW stat */
161  fsm->stats[FSM_NEW]++;
162 
163  /* Checks */
164  if (fsm->driver->start == NULL)
165  {
166  _dbg("'%s' has no start. Ooops! Breaking here now ...\n",
167  fsm->name);
168 
169  /* Exits inmediately */
170  fsm->stats[FSM_ERROR]++;
171  return 1;
172  }
173 
174  if (fsm->driver->stop == NULL)
175  {
176  _dbg("'%s' has no stop. FSM_CONFIG_AUTO_RESTART set\n",
177  fsm->name);
178 
179  /* Infinite loop for non-stoppable fsm */
180  fsm->config |= FSM_CONFIG_AUTO_RESTART;
181  }
182 
183  /* Configure */
184  fsm->next = fsm->driver->start;
185 
186  /* Pause, if rearm is set it'll run automagically in 2 cycles */
187  fsm->state = FSM_PAUSE;
188  break;
189 
190  /* 'Nothing to do' state. fsm possibly ended and no rearm bit was found */
191  case FSM_PAUSE:
192  _dbg("%s@FSM_PAUSE\n", fsm->name);
193 
194  /* Record FSM_PAUSE stat */
195  fsm->stats[FSM_PAUSE]++;
196 
197  /* Check if it should be ran */
198  if ((fsm->config & FSM_CONFIG_STARTED) &&
199  ((fsm->next == fsm->driver->start) ||
200  (fsm->config & FSM_CONFIG_AUTO_RESTART)))
201  {
202  _dbg("'%s' fsm is ready/restart bit is set. Run it!\n",
203  fsm->name);
204 
205  /* Armed fsm, run it again */
206  fsm->next = fsm->driver->start;
207  fsm->state = FSM_RUN;
208  }
209  break;
210 
211  /* Machine core, where the things happens */
212  case FSM_RUN:
213  _dbg("%s@FSM_RUN\n", fsm->name);
214 
215  /* Record FSM_RUN stat */
216  fsm->stats[FSM_RUN]++;
217 
218  /* fsm was ran and finished, and it also was rearmed */
219  if ((fsm->driver->stop != NULL) && (fsm->now == fsm->driver->stop))
220  {
221  _dbg("'%s' has been restarted\n", fsm->name);
222 
223  /* Ensure a valid start point */
224  fsm->next = fsm->driver->start;
225  }
226 
227  /* Evaluate next state validity */
228  if (fsm->next == NULL)
229  {
230  _dbg("'%s' fsm has no 'next' state\n", fsm->name);
231 
232  /* Error, try last? */
233  /* @todo Error for now */
234  fsm->state = FSM_ERROR;
235  return 1;
236  }
237 
238  /* Shift states and run next (only if new state) */
239  if (fsm->next != fsm->now)
240  {
241  fsm->last = fsm->now;
242  fsm->now = fsm->next;
243  }
244 
245  /* Ensure executed state set a valid 'next' state before return */
246  fsm->next = NULL;
247 
248  /* Executes state function */
249  if (fsm->now(fsm))
250  {
251  /* Error, break */
252  _dbg("'%s' executed, but something wrong happened\n",
253  fsm->name);
254 
255  /* @todo Error for now */
256  fsm->state = FSM_ERROR;
257  return 1;
258  }
259 
260  /* Capture if this fsm is terminated */
261  if ((fsm->driver->stop != NULL) && (fsm->now == fsm->driver->stop))
262  {
263  fsm->state = FSM_PAUSE;
264  }
265  break;
266 
267  /* You're in a little trouble if you are here */
268  case FSM_ERROR:
269  _dbg("%s@FSM_ERROR\n", fsm->name);
270  _dbg("'%s' fsm entered error state\n", fsm->name);
271  _dbg("- fsm->last .: %p\n", (void *)(size_t)fsm->last);
272  _dbg("- fsm->now ..: %p\n", (void *)(size_t)fsm->now);
273  _dbg("- fsm->next .: %p\n", (void *)(size_t)fsm->next);
274  _dbg("App needs to rearm it before continue\n");
275 
276  /* Record FSM_ERROR stat */
277  fsm->stats[FSM_ERROR]++;
278  break;
279 
280  /* You're in a big trouble if you are here */
281  default:
282  /* No stats here (Resultant is 'all' - 'sum of all of the rest') */
283  _dbg("'%s' went crazy, it shouldn't be here\n", fsm->name);
284 
285  /* @todo Error for now */
286  fsm->state = FSM_ERROR;
287  return 1;
288  break;
289  }
290 
291  return 0;
292 
293  /* Free _dbg() config */
294  #undef YCP_FNAME
295 }
