144 /* Initialize a *fsm_driver_simple* type FSM.
145  * Read `yacup/fsm/driver/simple.h` for complete information. */
146 int fsm_driver_simple(struct fsm *fsm)
147 {
148  /* Configure _dbg() */
149  #define YCP_FNAME "fsm_driver_simple"
150 
151  /* Create it static, as this will not change along the execution */
152  static struct fsm_driver this_driver =
153  {
154  .start = start,
155  .stop = stop
156  };
157 
158  /* Valid fsm? */
159  if (fsm == NULL)
160  {
161  _dbg("Direct calls are not recommended, read the doc\n");
162  return 1;
163  }
164 
165  /* This fsm will use data, do we have assigned storage for it? */
166  if (fsm->data == NULL)
167  {
168  _dbg("Invalid fsm data (fsm = %s)\n", fsm->name);
169  return 1;
170  }
171 
172  /* Assign default name, if not previously set */
173  if (fsm->name == NULL)
174  {
175  fsm->name = YCP_NAME;
176  }
177 
178  /* Fill this specific FSM data */
179  FSM_DATA(fsm)->cycles = 0;
180 
181  /* Assign the essential states */
182  fsm->driver = &this_driver;
183 
184  /* Let's go! */
185  return 0;
186 
187  /* Free _dbg() config */
188  #undef YCP_FNAME
189 }
