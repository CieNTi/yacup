35 /* Print `rb` information to STDOUT.
36  * Read `yacup/rb/debug.h` for complete information. */
37 void rb_print_info(struct rb *rb)
38 {
39  if ((rb == NULL) || (rb->op == NULL) || rb->op->validate(rb))
40  {
41  _dbg("Provided rb is not valid\n");
42  return;
43  }
44 
45  /* Required vars */
46  size_t idx = 0;
47 
48  /* Show data */
49  _dbg("[s: %3lu, h: %3lu, t: %3lu, of: %1u, len: %3lu]"
50  "[buf:",
51  rb->size,
52  rb->head,
53  rb->tail,
54  rb->head_of,
55  rb->op->len(rb)
56  );
57  for (idx = 0; idx < rb->size; idx++)
58  {
59  printf(" %02X", rb->buffer[idx]);
60  }
61  printf("]\n");
62  fflush(stdout);
63  return;
64 }
