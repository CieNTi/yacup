31 /* Print `rb` information to STDOUT.
32  * Read `yacup/rb/debug.h` for complete information. */
33 void rb_print_info(struct rb *rb)
34 {
35  /* Configure _dbg() */
36  #define YCP_FNAME "rb_print_info"
37 
38  if ((rb == NULL) || (rb->driver == NULL) || rb->driver->validate(rb))
39  {
40  _dbg("Provided rb is not valid\n");
41  return;
42  }
43 
44  /* Show parameters data */
45  _dbg("[s: %3lu, h: %3lu, t: %3lu, of: %1u, len: %3lu]\n",
46  rb->size,
47  rb->head,
48  rb->tail,
49  rb->head_of,
50  rb->driver->len(rb)
51  );
52 
53  /* Required vars to print a data table */
54  size_t idx_byte = 0;
55  size_t idx_line = 0;
56  #define RB_DBG_LINE_BYTES 24
57  #define __RB_DBG_TOTAL ((RB_DBG_LINE_BYTES * 3) + 10)
58  char debug_line[__RB_DBG_TOTAL] = { 0x00 };
59  char *debug_line_pt = NULL;
60 
61  /* Show top row with index */
62  debug_line_pt = debug_line;
63  for (idx_byte = 0; idx_byte < RB_DBG_LINE_BYTES; idx_byte++)
64  {
65  sprintf(debug_line_pt, " %02lX", idx_byte);
66  debug_line_pt += 3;
67  }
68  _dbg("[ |%s ]\n", debug_line);
69 
70  /* Show row with separator */
71  debug_line_pt = debug_line;
72  for (idx_byte = 0; idx_byte < RB_DBG_LINE_BYTES; idx_byte++)
73  {
74  sprintf(debug_line_pt, "~~~");
75  debug_line_pt += 3;
76  }
77  _dbg("[~~~~+%s~]\n", debug_line);
78 
79  /* Show rows with data */
80  for (idx_line = 0; (idx_line * RB_DBG_LINE_BYTES) < rb->size; idx_line++)
81  {
82  debug_line_pt = debug_line;
83  for (idx_byte = 0; idx_byte < RB_DBG_LINE_BYTES; idx_byte++)
84  {
85  if ((idx_line * RB_DBG_LINE_BYTES) + idx_byte < rb->size)
86  {
87  sprintf(debug_line_pt, "%c%02X",
88  (((idx_line * RB_DBG_LINE_BYTES) + idx_byte == rb->tail)?'[':
89  ((idx_line * RB_DBG_LINE_BYTES) + idx_byte == rb->head)?']':' '),
90  rb->buffer[(idx_line * RB_DBG_LINE_BYTES) + idx_byte]);
91  }
92  else
93  {
94  sprintf(debug_line_pt, " ~~");
95  }
96  debug_line_pt += 3;
97  }
98  _dbg("[ %02lX |%s ]\n", idx_line, debug_line);
99  }
100  return;
101 
102  /* Free _dbg() config */
103  #undef YCP_FNAME
104 }
