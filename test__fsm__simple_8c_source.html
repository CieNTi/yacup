52 int test_fsm_simple(int argc, const char* argv[])
53 {
54  /* Testbench vars */
55  #define MAX_CYCLES 20
56  uint32_t current_cycle = MAX_CYCLES;
57 
58  /* FSM data */
59  struct fsm_simple_data fsm_simple0_data;
60  struct fsm fsm_simple0 =
61  {
62  .name = "fsm_simple0",
63  .data = &fsm_simple0_data
64  };
65 
66  /* fsm (fsm_simple.c) */
67  _dbg("Setting up the fsm\n");
68  if (fsm_simple_setup(&fsm_simple0))
69  {
70  _dbg("- Cannot setup the FSM. ERROR\n");
71  return 1;
72  }
73  _dbg("- Ok\n");
74 
75  /* Basics: enable + auto-restart + start as soon as possible */
76  fsm_enable(&fsm_simple0);
77  fsm_set_loop(&fsm_simple0);
78  fsm_request_start(&fsm_simple0);
79 
80  /* fsm cycles */
81  fsm_print_info(&fsm_simple0);
82  while (current_cycle--)
83  {
84  /* Start cycle */
85  _dbg("Cycle #%02u\n", MAX_CYCLES - current_cycle);
86 
87  /* Test the fsm */
88  if (fsm_do_cycle(&fsm_simple0))
89  {
90  _dbg("Error when executing a fsm cycle\n");
91  fsm_print_info(&fsm_simple0);
92  fsm_print_stats(&fsm_simple0);
93  return 1;
94  }
95 
96  /* Finish cycle */
97  fflush(stdout);
98 
99  /* Simulate some time spent on the cycle */
100  nanosleep(&(struct timespec){ .tv_sec = 0, .tv_nsec = 100*1000000 }, NULL);
101  }
102 
103  /* Cya! */
104  fsm_print_stats(&fsm_simple0);
105  return 0;
106 }
