39 /* Checks if the ring-buffer is valid or not.
40  * Read `yacup/rb/op.h` for complete information. */
41 static int validate(struct rb *rb)
42 {
43  if (/* rb cannot be null */
44  (rb == NULL) ||
45  /* Buffer cannot be null */
46  (rb->buffer == NULL) ||
47  /* Size cannot be 0 */
48  (rb->size == 0) ||
49  /* Head has to be valid */
50  (rb->head >= rb->size) ||
51  /* Tail has to be valid */
52  (rb->tail >= rb->size) ||
53  /* Head >= tail */
54  (((rb->size * rb->head_of) + rb->head) < rb->tail))
55  {
56  _dbg("validate ~~~~~~~~~~~~~~~~~~~~~~\n");
57  _dbg("- rb cannot be null .....: %4s\n",
58  (rb == NULL)?"FAIL":"OK");
59  _dbg("- Buffer cannot be null .: %4s\n",
60  (rb->buffer == NULL)?"FAIL":"OK");
61  _dbg("- Size cannot be 0 ......: %4s\n",
62  (rb->size == 0)?"FAIL":"OK");
63  _dbg("- Head has to be valid ..: %4s\n",
64  (rb->head >= rb->size)?"FAIL":"OK");
65  _dbg("- Tail has to be valid ..: %4s\n",
66  (rb->tail >= rb->size)?"FAIL":"OK");
67  _dbg("- Head >= tail ..........: %4s\n",
68  (((rb->size * rb->head_of) + rb->head) < rb->tail)?"FAIL":"OK");
69  _dbg("~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\n");
70  return 1;
71  }
72 
73  /* Ok */
74  return 0;
75 }

77 /* Reset a ring-buffer by cleaning its content and making it empty.
78  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
79  * Read `yacup/rb/op.h` for complete information. */
80 static void reset(struct rb *rb)
81 {
82  /* Put properties to reset state */
83  rb->head = 0;
84  rb->tail = 0;
85  rb->head_of = 0;
86 
87  /* Clear data */
88  for (rb->head = 0; rb->head < rb->size; rb->head++)
89  {
90  rb->buffer[rb->head] = 0x00;
91  }
92  rb->head = 0;
93 
94  /* It is a void, just return */
95  return;
96 }

98 /* Add a byte to a ring-buffer head, overwritting if needed.
99  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
100  * Read `yacup/rb/op.h` for complete information. */
101 static int push(struct rb *rb, uint8_t byte)
102 {
103  /* Add the data, overwritting if needed */
104  rb->buffer[rb->head] = byte;
105 
106  if (rb->op->full(rb))
107  {
108  rb->tail++;
109  }
110  rb->head++;
111 
112  /* Normalize head index */
113  if (rb->head >= rb->size)
114  {
115  rb->head -= rb->size;
116  rb->head_of = 1;
117  }
118 
119  /* Normalize tail index */
120  if (rb->tail >= rb->size)
121  {
122  rb->tail -= rb->size;
123  }
124 
125  /* Ok! */
126  return 0;
127 }

129 /* Read and delete a byte from a ring-buffer tail.
130  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
131  * Read `yacup/rb/op.h` for complete information. */
132 static int pull(struct rb *rb, uint8_t *byte)
133 {
134  if (rb->op->len(rb) == 0)
135  {
136  _dbg("pull: Pulling from empty rb not allowed\n");
137  return 1;
138  }
139 
140  /* Extract the data */
141  *byte = rb->buffer[rb->tail];
142  rb->buffer[rb->tail] = 0x00;
143  rb->tail++;
144 
145  /* Normalize tail index */
146  if (rb->tail >= rb->size)
147  {
148  rb->tail -= rb->size;
149  rb->head_of = 0;
150  }
151 
152  /* Ok! */
153  return 0;
154 }
