27 /* Initializes a command engine channel using driver initialization functions
28  * Read `yacup/ce/channel.h` for complete information. */
29 int ce_channel_init(struct ce_channel *channel,
30  int (*channels_codec_driver_init)(struct ce_codec *),
31  int (*channels_rb_driver_init)(struct rb *))
32 {
33  /* Configure _dbg() */
34  #define YCP_FNAME "ce_channel_init"
35 
36  if (/* Invalid channel? */
37  (channel == NULL) ||
38  /* No commands? */
39  (channel->command_set == NULL) ||
40  /* Defined buffer driver? */
41  (channels_rb_driver_init == NULL) ||
42  /* Defined codec driver? */
43  (channels_codec_driver_init == NULL) ||
44  /* Valid data rb? */
45  rb_init(&channel->data, channels_rb_driver_init) ||
46  /* Valid message rb? */
47  rb_init(&channel->message, channels_rb_driver_init) ||
48  /* Valid codec? */
49  ce_codec_init(&channel->codec, channels_codec_driver_init))
50  {
51  _dbg("Invalid ce_channel or driver init function\n");
52  return 1;
53  }
54 
55  /* Ok! */
56  _dbg("Channel initialized successfully\n");
57  return 0;
58 
59  /* Free _dbg() config */
60  #undef YCP_FNAME
61 }
