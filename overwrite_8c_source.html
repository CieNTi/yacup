28 /* Checks if the ring-buffer is valid or not.
29  * Read `yacup/rb/driver.h` for complete information. */
30 static int validate(struct rb *rb)
31 {
32  /* Configure _dbg() */
33  #define YCP_FNAME "validate"
34 
35  if (/* rb cannot be null */
36  (rb == NULL) ||
37  /* Buffer cannot be null */
38  (rb->buffer == NULL) ||
39  /* Size cannot be 0 */
40  (rb->size == 0) ||
41  /* Head has to be valid */
42  (rb->head >= rb->size) ||
43  /* Tail has to be valid */
44  (rb->tail >= rb->size) ||
45  /* Head >= tail */
46  (((rb->size * rb->head_of) + rb->head) < rb->tail))
47  {
48  _dbg("~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\n");
49  _dbg("- rb cannot be null .....: %4s\n",
50  (rb == NULL)?"FAIL":"OK");
51  _dbg("- Buffer cannot be null .: %4s\n",
52  (rb->buffer == NULL)?"FAIL":"OK");
53  _dbg("- Size cannot be 0 ......: %4s\n",
54  (rb->size == 0)?"FAIL":"OK");
55  _dbg("- Head has to be valid ..: %4s\n",
56  (rb->head >= rb->size)?"FAIL":"OK");
57  _dbg("- Tail has to be valid ..: %4s\n",
58  (rb->tail >= rb->size)?"FAIL":"OK");
59  _dbg("- Head >= tail ..........: %4s\n",
60  (((rb->size * rb->head_of) + rb->head) < rb->tail)?"FAIL":"OK");
61  _dbg("~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~\n");
62  return 1;
63  }
64 
65  /* Ok */
66  return 0;
67 
68  /* Free _dbg() config */
69  #undef YCP_FNAME
70 }

72 /* Reset a ring-buffer by cleaning its content and making it empty.
73  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
74  * Read `yacup/rb/driver.h` for complete information. */
75 static void reset(struct rb *rb)
76 {
77  /* Put properties to reset state */
78  rb->head = 0;
79  rb->tail = 0;
80  rb->head_of = 0;
81 
82  /* Clear data */
83  for (rb->head = 0; rb->head < rb->size; rb->head++)
84  {
85  rb->buffer[rb->head] = 0x00;
86  }
87  rb->head = 0;
88 
89  /* It is a void, just return */
90  return;
91 }

93 /* Add a byte to a ring-buffer head, overwritting if needed.
94  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
95  * Read `yacup/rb/driver.h` for complete information. */
96 static int push(struct rb *rb, uint8_t byte)
97 {
98  /* Add the data, overwritting if needed */
99  rb->buffer[rb->head] = byte;
100 
101  if (rb->driver->full(rb))
102  {
103  rb->tail++;
104  }
105  rb->head++;
106 
107  /* Normalize head index */
108  if (rb->head >= rb->size)
109  {
110  rb->head -= rb->size;
111  rb->head_of = 1;
112  }
113 
114  /* Normalize tail index */
115  if (rb->tail >= rb->size)
116  {
117  rb->tail -= rb->size;
118  }
119 
120  /* Ok! */
121  return 0;
122 }

124 /* Read and delete a byte from a ring-buffer tail.
125  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
126  * Read `yacup/rb/driver.h` for complete information. */
127 static int pull(struct rb *rb, uint8_t *byte)
128 {
129  /* Configure _dbg() */
130  #define YCP_FNAME "pull"
131 
132  if (rb->driver->len(rb) == 0)
133  {
134  _dbg("Pulling from empty rb not allowed\n");
135  return 1;
136  }
137 
138  /* Extract the data */
139  *byte = rb->buffer[rb->tail];
140  rb->buffer[rb->tail] = 0x00;
141  rb->tail++;
142 
143  /* Normalize tail index */
144  if (rb->tail >= rb->size)
145  {
146  rb->tail -= rb->size;
147  rb->head_of = 0;
148  }
149 
150  /* Ok! */
151  return 0;
152 
153  /* Free _dbg() config */
154  #undef YCP_FNAME
155 }

157 /* Write a byte by position on a ring-buffer, without updating head/tail.
158  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
159  * Read `yacup/rb/driver.h` for complete information. */
160 static int write(struct rb *rb, uint8_t byte, size_t position)
161 {
162  /* Use position as index over tail, with overflow management */
163  rb->buffer[(rb->tail + position) % rb->size] = byte;
164 
165  /* Ok! */
166  return 0;
167 }
168 
169 /* Read a byte by position from a ring-buffer, without deleting it.
170  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
171  * Read `yacup/rb/driver.h` for complete information. */
172 static int read(struct rb *rb, uint8_t *byte, size_t position)
173 {
174  /* Use position as index over tail, with overflow management */
175  *byte = rb->buffer[(rb->tail + position) % rb->size];
176 
177  /* Ok! */
178  return 0;
179 }

181 /* Returns max available size of a ring-buffer buffer.
182  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
183  * Read `yacup/rb/driver.h` for complete information. */
184 static size_t size(struct rb *rb)
185 {
186  return rb->size;

189 /* Returns available data size inside a ring-buffer.
190  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
191  * Read `yacup/rb/driver.h` for complete information. */
192 static size_t len(struct rb *rb)
193 {
194  return (((rb->size * rb->head_of) + rb->head) - rb->tail);
195 }

197 /* Checks if a ring-buffer is full or not.
198  * WARNING: Assumed `rb/rb.c` pre-validation. Not safe as direct call!
199  * Read `yacup/rb/driver.h` for complete information. */
200 static uint8_t full(struct rb *rb)
201 {
202  return ((rb->head_of == 1) && (rb->head == rb->tail));
203 }

205 /* Initialize a `overwrite` type RB.
206  * Read `yacup/rb/driver.h` for complete information. */
207 int rb_driver_overwrite(struct rb *rb)
208 {
209  /* Configure _dbg() */
210  #define YCP_FNAME "rb_driver_overwrite"
211 
212  /* Create it static, as this will not change along the execution */
213  static struct rb_driver this_driver =
214  {
215  .name = YCP_NAME,
216  .validate = validate,
217  .reset = reset,
218  .push = push,
219  .pull = pull,
220  .write = write,
221  .read = read,
222  .size = size,
223  .len = len,
224  .full = full
225  };
226 
227  /* Valid rb? */
228  if (rb == NULL)
229  {
230  _dbg("Direct calls not recommended, read the doc\n");
231  return 1;
232  }
233 
234  /* Ok assign the operations */
235  rb->driver = &this_driver;
236 
237  /* And return with success */
238  return 0;
239 
240  /* Free _dbg() config */
241  #undef YCP_FNAME
242 }
