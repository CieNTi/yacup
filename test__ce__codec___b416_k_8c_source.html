33 int encode_decode_check_by_type(struct ce_codec *codec,
34  enum ce_data_type type,
35  struct rb *rb)
36 {
37  /* Configure _dbg() */
38  #define YCP_FNAME "encode_decode_check_by_type"
39 
40  /* All data types, source and destination flavours */
41  uint8_t src_uint8_t = 100; uint8_t dst_uint8_t = 0;
42  int8_t src_int8_t = -100; int8_t dst_int8_t = 0;
43  uint16_t src_uint16_t = 300; uint16_t dst_uint16_t = 0;
44  int16_t src_int16_t = -300; int16_t dst_int16_t = 0;
45  uint32_t src_uint32_t = 100000; uint32_t dst_uint32_t = 0;
46  int32_t src_int32_t = -100000; int32_t dst_int32_t = 0;
47  uint64_t src_uint64_t = 4300000000; uint64_t dst_uint64_t = 0;
48  int64_t src_int64_t = -4300000000; int64_t dst_int64_t = 0;
49  float src_float = -1.23456789012; float dst_float = 0.0;
50  double src_double[3] = {-1.2, 0.0, 1.2}; double dst_double[3] = {0.0};
51  /* Pointers to them, both fashions */
52  void *src_pt = NULL; void *dst_pt = NULL;
53  size_t len_data = 0;
54 
55  /* Branch on type to assign valid pointers */
56  switch(type)
57  {
58  case CE_DATA_UINT8_T:
59  _dbg("Selected data type CE_DATA_UINT8_T = %lu\n", (size_t)type);
60  src_pt = &src_uint8_t; dst_pt = &dst_uint8_t; len_data = 1;
61  break;
62  case CE_DATA_INT8_T:
63  _dbg("Selected data type CE_DATA_INT8_T = %lu\n", (size_t)type);
64  src_pt = &src_int8_t; dst_pt = &dst_int8_t; len_data = 1;
65  break;
66  case CE_DATA_UINT16_T:
67  _dbg("Selected data type CE_DATA_UINT16_T = %lu\n", (size_t)type);
68  src_pt = &src_uint16_t; dst_pt = &dst_uint16_t; len_data = 1;
69  break;
70  case CE_DATA_INT16_T:
71  _dbg("Selected data type CE_DATA_INT16_T = %lu\n", (size_t)type);
72  src_pt = &src_int16_t; dst_pt = &dst_int16_t; len_data = 1;
73  break;
74  case CE_DATA_UINT32_T:
75  _dbg("Selected data type CE_DATA_UINT32_T = %lu\n", (size_t)type);
76  src_pt = &src_uint32_t; dst_pt = &dst_uint32_t; len_data = 1;
77  break;
78  case CE_DATA_INT32_T:
79  _dbg("Selected data type CE_DATA_INT32_T = %lu\n", (size_t)type);
80  src_pt = &src_int32_t; dst_pt = &dst_int32_t; len_data = 1;
81  break;
82  case CE_DATA_UINT64_T:
83  _dbg("Selected data type CE_DATA_UINT64_T = %lu\n", (size_t)type);
84  src_pt = &src_uint64_t; dst_pt = &dst_uint64_t; len_data = 1;
85  break;
86  case CE_DATA_INT64_T:
87  _dbg("Selected data type CE_DATA_INT64_T = %lu\n", (size_t)type);
88  src_pt = &src_int64_t; dst_pt = &dst_int64_t; len_data = 1;
89  break;
90  case CE_DATA_FLOAT:
91  _dbg("Selected data type CE_DATA_FLOAT = %lu\n", (size_t)type);
92  src_pt = &src_float; dst_pt = &dst_float; len_data = 1;
93  break;
94  case CE_DATA_DOUBLE:
95  _dbg("Selected data type CE_DATA_DOUBLE = %lu\n", (size_t)type);
96  src_pt = &src_double; dst_pt = &dst_double; len_data = 3;
97  break;
98  default:
99  _dbg("Unrecognized type %lu\n", (size_t)type);
100  return 1;
101  }
102 
103  /* Encode selected variable */
104  _dbg("Encoding type %lu from address %p\n", (size_t)type, src_pt);
105  if(codec->encode.data(type, src_pt, len_data, rb) == 0)
106  {
107  _dbg("- Cannot encode the data. ERROR\n");
108  return 1;
109  }
110  _dbg("- Ok\n");
111 
112  /* Decode a selected variable */
113  _dbg("Decoding type %lu to address %p\n", (size_t)type, dst_pt);
114  if(codec->decode.data(rb, type, dst_pt, len_data) == 0)
115  {
116  _dbg("- Cannot decode the data. ERROR\n");
117  return 1;
118  }
119  _dbg("- Ok\n");
120 
121  /* Branch on type to check each returned value */
122  uint8_t res = 0;
123  switch(type)
124  {
125  case CE_DATA_UINT8_T:
126  _dbg("Checking data type CE_DATA_UINT8_T = %lu\n", (size_t)type);
127  res = (src_uint8_t == dst_uint8_t);
128  _dbg("%u == %u?\n", src_uint8_t, dst_uint8_t);
129  break;
130  case CE_DATA_INT8_T:
131  _dbg("Checking data type CE_DATA_INT8_T = %lu\n", (size_t)type);
132  res = (src_int8_t == dst_int8_t);
133  _dbg("%i == %i?\n", src_int8_t, dst_int8_t);
134  break;
135  case CE_DATA_UINT16_T:
136  _dbg("Checking data type CE_DATA_UINT16_T = %lu\n", (size_t)type);
137  res = (src_uint16_t == dst_uint16_t);
138  _dbg("%u == %u?\n", src_uint16_t, dst_uint16_t);
139  break;
140  case CE_DATA_INT16_T:
141  _dbg("Checking data type CE_DATA_INT16_T = %lu\n", (size_t)type);
142  res = (src_int16_t == dst_int16_t);
143  _dbg("%i == %i?\n", src_int16_t, dst_int16_t);
144  break;
145  case CE_DATA_UINT32_T:
146  _dbg("Checking data type CE_DATA_UINT32_T = %lu\n", (size_t)type);
147  res = (src_uint32_t == dst_uint32_t);
148  _dbg("%u == %u?\n", src_uint32_t, dst_uint32_t);
149  break;
150  case CE_DATA_INT32_T:
151  _dbg("Checking data type CE_DATA_INT32_T = %lu\n", (size_t)type);
152  res = (src_int32_t == dst_int32_t);
153  _dbg("%i == %i?\n", src_int32_t, dst_int32_t);
154  break;
155  case CE_DATA_UINT64_T:
156  _dbg("Checking data type CE_DATA_UINT64_T = %lu\n", (size_t)type);
157  res = (src_uint64_t == dst_uint64_t);
158  _dbg("%lu == %lu?\n", src_uint64_t, dst_uint64_t);
159  break;
160  case CE_DATA_INT64_T:
161  _dbg("Checking data type CE_DATA_INT64_T = %lu\n", (size_t)type);
162  res = (src_int64_t == dst_int64_t);
163  _dbg("%li == %li?\n", src_int64_t, dst_int64_t);
164  break;
165  case CE_DATA_FLOAT:
166  _dbg("Checking data type CE_DATA_FLOAT = %lu\n", (size_t)type);
167  res = (src_float == dst_float);
168  _dbg("%f == %f?\n", src_float, dst_float);
169  break;
170  case CE_DATA_DOUBLE:
171  _dbg("Checking data type CE_DATA_DOUBLE = %lu\n", (size_t)type);
172  res = ((src_double[0] == dst_double[0]) &&
173  (src_double[1] == dst_double[1]) &&
174  (src_double[2] == dst_double[2]));
175  _dbg("%f == %f?\n", src_double[0], dst_double[0]);
176  _dbg("%f == %f?\n", src_double[1], dst_double[1]);
177  _dbg("%f == %f?\n", src_double[2], dst_double[2]);
178  break;
179  default:
180  _dbg("What? How did you arrive here bro? SUPER FAIL\n");
181  return 1;
182  }
183 
184  if (res == 1)
185  {
186  _dbg(":) Valid comparison!\n");
187  }
188  else
189  {
190  _dbg(":( Invalid comparison!\n");
191  return 1;
192  }
193 
194  /* Ok! */
195  return 0;
196 
197  /* Free _dbg() config */
198  #undef YCP_FNAME
199 }

216 int test_ce_codec_B416K(int argc, const char* argv[])
217 {
218  /* Configure _dbg() */
219  #define YCP_FNAME "test_ce_codec_B416K"
220 
221  _dbg("Hi! from "__FILE__"\n");
222 
223  /*
224  * HIGH-LEVEL: Inside test/app
225  */
226  /* What a codec needs:
227  * - one rb where to push the encoded data
228  * - one rb where to save a message
229  */
230  #define TEST_CE_CODEC_B416K_DATA 512
231  uint8_t buffer_data[TEST_CE_CODEC_B416K_DATA];
232  struct rb rb_data =
233  {
234  .buffer = buffer_data,
235  .size = TEST_CE_CODEC_B416K_DATA
236  };
237 
238  /* Create rb_data using overwrite driver */
239  _dbg("Initializing the 'rb_data' ring-buffer\n");
240  if (rb_init(&rb_data, rb_driver_overwrite))
241  {
242  _dbg("Cannot initialize rb_data\n");
243  return 1;
244  }
245 
246  #define TEST_CE_CODEC_B416K_MESSAGE 1024
247  uint8_t buffer_message[TEST_CE_CODEC_B416K_MESSAGE];
248  struct rb rb_message =
249  {
250  .buffer = buffer_message,
251  .size = TEST_CE_CODEC_B416K_MESSAGE
252  };
253 
254  /* Create rb_message using overwrite driver */
255  _dbg("Initializing the 'rb_message' ring-buffer\n");
256  if (rb_init(&rb_message, rb_driver_overwrite))
257  {
258  _dbg("Cannot initialize rb_message\n");
259  return 1;
260  }
261 
262  /*
263  * MID-LOW-LEVEL: Calls to codec functions inside a `ce`
264  */
265  /* Set by `ce_init` */
266  struct ce_codec codec0 = { 0x00 };
267  _dbg("Printing uninitialized 'codec0'\n");
268  ce_codec_print_info(&codec0);
269 
270  /* Initialize codec found at `B416K.c` */
271  _dbg("Initializing the `B416K` codec\n");
272  if (ce_codec_init(&codec0, ce_codec_B416K))
273  {
274  _dbg("- Cannot initialize the codec. ERROR\n");
275  return 1;
276  }
277  _dbg("- Ok\n");
278  _dbg("Printing initialized 'codec0', data and message not be NULL\n");
279  ce_codec_print_info(&codec0);
280 
281  /* Encode/decode and Compare data */
282  _dbg("Per type: Encode -> Decode -> Compare\n");
283  if(encode_decode_check_by_type(&codec0, CE_DATA_UINT8_T, &rb_data) ||
284  encode_decode_check_by_type(&codec0, CE_DATA_INT8_T, &rb_data) ||
285  encode_decode_check_by_type(&codec0, CE_DATA_UINT16_T, &rb_data) ||
286  encode_decode_check_by_type(&codec0, CE_DATA_INT16_T, &rb_data) ||
287  encode_decode_check_by_type(&codec0, CE_DATA_UINT32_T, &rb_data) ||
288  encode_decode_check_by_type(&codec0, CE_DATA_INT32_T, &rb_data) ||
289  encode_decode_check_by_type(&codec0, CE_DATA_UINT64_T, &rb_data) ||
290  encode_decode_check_by_type(&codec0, CE_DATA_INT64_T, &rb_data) ||
291  encode_decode_check_by_type(&codec0, CE_DATA_FLOAT, &rb_data) ||
292  encode_decode_check_by_type(&codec0, CE_DATA_DOUBLE, &rb_data))
293  {
294  _dbg("- A step failed. ERROR\n");
295  return 1;
296  }
297  _dbg("- Ok\n");
298 
299  /* Encode/decode and Compare messages */
300  #define __DATA_LEN 400
301  uint8_t some_src_var[__DATA_LEN] = { 0xBE, 0xEF };
302  uint8_t some_dst_var[__DATA_LEN] = { 0x00 };
303 
304  /* Encode data into a data buffer */
305  _dbg("Preparing a data packet with %u bytes\n", __DATA_LEN);
306  if(codec0.encode.data(CE_DATA_UINT8_T,
307  some_src_var, __DATA_LEN,
308  &rb_data) == 0)
309  {
310  _dbg("- Cannot encode the data. ERROR\n");
311  return 1;
312  }
313  _dbg("- Ok\n");
314 
315  _dbg("Data buffer content:\n");
316  rb_print_info(&rb_data);
317 
318  /* Encode data buffer/packet into a message */
319  _dbg("Encoding data packet into a message\n");
320  if(codec0.encode.message(&rb_data, &rb_message))
321  {
322  _dbg("- Cannot encode the message. ERROR\n");
323  return 1;
324  }
325  _dbg("- Ok\n");
326 
327  /* Simulate invalid data before a valid message */
328  _dbg("Adding invalid data before message, acting as comm error data\n");
329  size_t orig_len = rb_len(&rb_message);
330  uint8_t a_u8 = 0;
331  rb_push(&rb_message, 0xAE);
332  rb_push(&rb_message, 0xBE);
333  rb_push(&rb_message, 0xCE);
334  rb_push(&rb_message, 0xDE);
335  rb_push(&rb_message, 0xEE);
336  for (;orig_len > 0; orig_len--)
337  {
338  if (rb_pull(&rb_message, &a_u8) ||
339  rb_push(&rb_message, a_u8))
340  {
341  /* Ooops! */
342  return 1;
343  }
344  }
345 
346  _dbg("Message buffer content:\n");
347  rb_print_info(&rb_message);
348 
349  /* Reset rb_data to be used as a receptacle for decoded data */
350  rb_reset(&rb_data);
351  _dbg("Data buffer content after reset:\n");
352  rb_print_info(&rb_data);
353 
354  /* Decode a message extracting its data packet */
355  _dbg("Decoding data packet from a message\n");
356  if(codec0.decode.message(&rb_message, &rb_data))
357  {
358  _dbg("- Cannot decode the message. ERROR\n");
359  return 1;
360  }
361  _dbg("- Ok\n");
362 
363  /* Decoded data buffer */
364  _dbg("Data buffer content:\n");
365  rb_print_info(&rb_data);
366 
367  /* Decode the real data from the data buffer */
368  _dbg("Decoding a data packet of %u bytes\n", __DATA_LEN);
369  if(codec0.decode.data(&rb_data, CE_DATA_UINT8_T,
370  some_dst_var, __DATA_LEN) == 0)
371  {
372  _dbg("- Cannot decode the data. ERROR\n");
373  return 1;
374  }
375  _dbg("- Ok\n");
376 
377  /* Final check */
378  _dbg("Comparing all %u bytes (source vs. destination)\n", __DATA_LEN);
379  size_t idx = 0;
380  for (idx = 0; idx < __DATA_LEN; idx++)
381  {
382  if (some_src_var[idx] != some_dst_var[idx])
383  {
384  _dbg(":( Invalid comparison!\n");
385  return 1;
386  }
387  }
388  _dbg(":) Valid comparison!\n");
389 
390  /* Cya! */
391  return 0;
392 
393  /* Free _dbg() config */
394  #undef YCP_FNAME
395 }
