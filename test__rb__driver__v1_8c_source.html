120

133  {

148  }

150  rb_print_info(rb0);
151  }
152

156  {
157  _dbg("- Expected: cienti. ERROR\n");
158  return 1;
159  }
160  _dbg("- OK\n");
161 
162  /* Reset it! */
163  rb_reset(rb0);
164 
165  /* Debug it! */
166  rb_print_info(rb0);
167
