70 {
71  /* Configure _dbg() */

73 
74  if (FSM_DATA(fsm)->ce == NULL)

113 static int s_wait_cmd(struct fsm *fsm)
114 {
115  /* Configure _dbg() */
116  #define YCP_FNAME "s_wait_cmd"
117 
118  /* Default next state */
119  fsm->next = s_wait_cmd;
120 
121  /* Do you want to send? */
122  if (FSM_DATA(fsm)->request_to_send == 1)
123  {
124  FSM_DATA(fsm)->message_sent = 0;
125  fsm->next = s_encode;
126  }
127  /* Or do you want to receive? */
128  if (FSM_DATA(fsm)->request_to_receive == 1)
129  {
130  FSM_DATA(fsm)->message_decoded = 0;
131  fsm->next = s_receive;
132  }
133  return 0;

136  #undef YCP_FNAME

150 static int s_encode(struct fsm *fsm)
151 {
152  /* Configure _dbg() */
153  #define YCP_FNAME "s_encode"
154 
155  /* Default next state */
156  fsm->next = s_send;
157 
158  /* Encode the command as a data payload */
159  if (FSM_DATA(fsm)->ce->out.codec.encode.command(FSM_DATA(fsm)->command,
160  FSM_DATA(fsm)->argument,
161  &(FSM_DATA(fsm)->ce->out.data)))
162  {
163  fsm->next = s_error;
164 
165  _dbg("Cannot encode command to payload\n");
166  return 1;
167  }
168 
169  /* Print output buffer before using the data */
170  _dbg("Data buffer (should be still filled, but it will be erased now)\n");
171  rb_print_info(&(FSM_DATA(fsm)->ce->out.data));
172 
173  /* Encode the data payload as a message */
174  if (FSM_DATA(fsm)->ce->out.codec.encode.message(
175  &(FSM_DATA(fsm)->ce->out.data),
176  &(FSM_DATA(fsm)->ce->out.message)))
177  {
178  fsm->next = s_error;
179 
180  _dbg("Cannot encode payload to message\n");
181  return 1;
182  }
183 
184  /* Encoded, so send it now */
185  return 0;
186 
187  /* Free _dbg() config */
188  #undef YCP_FNAME

203 {

209 
210  /* Decode the message into data payload */
211  if (FSM_DATA(fsm)->ce->in.codec.decode.message(&(FSM_DATA(fsm)->ce->in.message),
212  &(FSM_DATA(fsm)->ce->in.data)))
213  {

215 
216  _dbg("Cannot decode message to payload\n");
217  return 1;
218  }
219 
220  /* Print input buffer before using the data */
221  _dbg("Data buffer (should be still filled, but it will be erased now)\n");
222  rb_print_info(&(FSM_DATA(fsm)->ce->in.data));
223 
224  /* Decode the payload into a command (saving data into listener argument, if
225  * present */
226  if (FSM_DATA(fsm)->ce->in.codec.decode.command(&(FSM_DATA(fsm)->ce->in.data),
227  FSM_DATA(fsm)->ce->in.command_set,

229  {
230  fsm->next = s_error;
231 
232  _dbg("Cannot decode payload to command\n");
233  return 1;
234  }
235 
236  /* Flag it and go */
237  FSM_DATA(fsm)->message_decoded = 1;
238 
239  /* Decoded, so go back now */
240  return 0;

243  #undef YCP_FNAME

406 /* Initialize a *ce_driver_faf* type *ce_driver*.
407  * Read `yacup/ce/driver/fire-and-forget.h` for complete information. */
408 static int send_command(struct ce *ce,
409  struct ce_command *command,
410  struct ce_command_argument *argument[])
411 {
412  /* Configure _dbg() */
413  #define YCP_FNAME "send_command"
414 
415  /* Command is already validated, so we can trust on:
416  * - arguments type, checked by ce_command_validate()
417  * - arguments type, ensured storage as per ce_command_argument declaration
418  */
419  _dbg("FAF is sending command '%s' (0x%02lX)\n", command->name, command->id);
420 
421  struct timespec time_to_finish = { 0x00 };
422  struct timespec time_after = { 0x00 };
423 
424  /* Calculate when the timeout will happen */
425  if(clock_gettime(CLOCK_REALTIME, &time_to_finish))
426  {
427  time_to_finish.tv_sec = 0;
428  time_to_finish.tv_nsec = 0;
429  }
430  time_to_finish.tv_sec += FSM_DATA(&ce->driver.fsm)->send_timeout.tv_sec;
431  time_to_finish.tv_nsec += FSM_DATA(&ce->driver.fsm)->send_timeout.tv_nsec;
432  time_to_finish.tv_sec += time_to_finish.tv_nsec / 1000000000;
433  time_to_finish.tv_nsec = time_to_finish.tv_nsec % 1000000000;
434 
435  FSM_DATA(&ce->driver.fsm)->message_sent = 0;
436  do
437  {
438  /* Once requests are accepted, put the command and go! */
439  if (ce->driver.fsm.next == s_wait_cmd)
440  {
441  FSM_DATA(&ce->driver.fsm)->command = command;
442  FSM_DATA(&ce->driver.fsm)->argument = argument;
443  FSM_DATA(&ce->driver.fsm)->request_to_send = 1;
444  }
445  if (fsm_do_cycle(&ce->driver.fsm))
446  {
447  _dbg("Error when executing a fsm cycle\n");
448  fsm_print_info(&ce->driver.fsm);
449  fsm_print_stats(&ce->driver.fsm);
450  return 1;
451  }
452  if ((ce->driver.fsm.now == s_stop) &&
453  (ce->driver.fsm.next == s_start))
454  {
455  if (FSM_DATA(&ce->driver.fsm)->message_sent == 1)
456  {
457  /* Sent! */
458  return 0;
459  }
460  /* Not sent! */
461  _dbg("Cannot send the command\n");
462  return 1;
463  }
464 
465  /* Simulate some time spent on the cycle */
466  nanosleep(&(FSM_DATA(&ce->driver.fsm)->fsm_delay), NULL);
467  }
468  while ((clock_gettime(CLOCK_REALTIME, &time_after) == 0) &&
469  ((time_after.tv_sec < time_to_finish.tv_sec) ||
470  ((time_after.tv_sec == time_to_finish.tv_sec) &&
471  (time_after.tv_nsec < time_to_finish.tv_nsec))));
472 
473  /* Ouch! */
474  _dbg("Timeout\n");
475  return 1;
476 
477  /* Free _dbg() config */
478  #undef YCP_FNAME
479 }

482 /* Initialize a *ce_driver_faf* type *ce_driver*.
483  * Read `yacup/ce/driver/fire-and-forget.h` for complete information. */
484 static int wait_command(struct ce *ce,
485  size_t *id,
486  struct ce_command_argument **argument[])
487 {
488  /* Configure _dbg() */
489  #define YCP_FNAME "wait_command"
490 
491  /* Command is already validated, so we can trust on:
492  * - arguments type, checked by ce_command_validate()
493  * - arguments type, ensured storage as per ce_command_argument declaration
494  */
495  _dbg("FAF is waiting commands\n");
496 
497  struct timespec time_to_finish = { 0x00 };
498  struct timespec time_after = { 0x00 };
499 
500  /* Calculate when the timeout will happen */
501  if(clock_gettime(CLOCK_REALTIME, &time_to_finish))
502  {
503  time_to_finish.tv_sec = 0;
504  time_to_finish.tv_nsec = 0;
505  }
506  time_to_finish.tv_sec += FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_sec;
507  time_to_finish.tv_nsec += FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_nsec;
508  time_to_finish.tv_sec += time_to_finish.tv_nsec / 1000000000;
509  time_to_finish.tv_nsec = time_to_finish.tv_nsec % 1000000000;
510 
511  FSM_DATA(&ce->driver.fsm)->message_decoded = 0;
512  do
513  {
514  if (ce->driver.fsm.next == s_wait_cmd)
515  {
516  FSM_DATA(&ce->driver.fsm)->command = NULL;
517  FSM_DATA(&ce->driver.fsm)->argument = NULL;
518  FSM_DATA(&ce->driver.fsm)->request_to_receive = 1;
519  }
520  if (fsm_do_cycle(&ce->driver.fsm))
521  {
522  _dbg("Error when executing a fsm cycle\n");
523  fsm_print_info(&ce->driver.fsm);
524  fsm_print_stats(&ce->driver.fsm);
525  return 1;
526  }
527  if ((ce->driver.fsm.now == s_stop) &&
528  (ce->driver.fsm.next == s_start))
529  {
530  if (FSM_DATA(&ce->driver.fsm)->message_decoded == 1)
531  {
532  _dbg("Here, or at s_decode, listener->listener can be called\n");
533  /* Fill id and arguments of caller */
534  *id = FSM_DATA(&ce->driver.fsm)->command->id;
535  *argument = NULL;
536  if (FSM_DATA(&ce->driver.fsm)->command->listener != NULL)
537  {
538  *argument = FSM_DATA(&ce->driver.fsm)->command->listener->argument;
539  }
540  /* Received! */
541  return 0;
542  }
543  /* Not sent! */
544  _dbg("Cannot decode the command\n");
545  return 1;
546  }
547 
548  /* Simulate some time spent on the cycle */
549  nanosleep(&(FSM_DATA(&ce->driver.fsm)->fsm_delay), NULL);
550  }
551  while ((clock_gettime(CLOCK_REALTIME, &time_after) == 0) &&
552  ((time_after.tv_sec < time_to_finish.tv_sec) ||
553  ((time_after.tv_sec == time_to_finish.tv_sec) &&
554  (time_after.tv_nsec < time_to_finish.tv_nsec))));
555 
556  /* Ouch! */
557  _dbg("Timeout\n");
558  return 1;
559 
560  /* Free _dbg() config */
561  #undef YCP_FNAME
562 }

565 /* Initialize a command engine
566  * Read `yacup/ce/driver/fire-and-forget.h` for complete information. */
567 int ce_driver_faf(struct ce *ce)
568 {
569  /* Configure _dbg() */
570  #define YCP_FNAME "ce_driver_faf"
571 
572  if (/* Valid engine? */
573  ce == NULL ||
574  /* Valid required fsm data */
575  ce->driver.fsm.data == NULL)
576  {
577  _dbg("Invalid command engine or required data\n");
578  return 1;
579  }
580 
581  /* Assign fsm external variables */
582  FSM_DATA(&ce->driver.fsm)->ce = ce;
583 
584  /* Set cycle delay if not */
585  if ((FSM_DATA(&ce->driver.fsm)->fsm_delay.tv_sec == 0) &&
586  (FSM_DATA(&ce->driver.fsm)->fsm_delay.tv_nsec == 0))
587  {
588  FSM_DATA(&ce->driver.fsm)->fsm_delay.tv_sec =
589  (CE_DRIVER_FAF_FSM_DELAY_MS / 1000);
590  FSM_DATA(&ce->driver.fsm)->fsm_delay.tv_nsec =
591  (CE_DRIVER_FAF_FSM_DELAY_MS % 1000) * 1000000;
592  }
593 
594  /* Set cycle delay if not */
595  if ((FSM_DATA(&ce->driver.fsm)->send_timeout.tv_sec == 0) &&
596  (FSM_DATA(&ce->driver.fsm)->send_timeout.tv_nsec == 0))
597  {
598  FSM_DATA(&ce->driver.fsm)->send_timeout.tv_sec =
599  (CE_DRIVER_FAF_SEND_TIMEOUT_MS / 1000);
600  FSM_DATA(&ce->driver.fsm)->send_timeout.tv_nsec =
601  (CE_DRIVER_FAF_SEND_TIMEOUT_MS % 1000) * 1000000;
602  }
603 
604  /* Set cycle delay if not */
605  if ((FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_sec == 0) &&
606  (FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_nsec == 0))
607  {
608  FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_sec =
609  (CE_DRIVER_FAF_RECEIVE_TIMEOUT_MS / 1000);
610  FSM_DATA(&ce->driver.fsm)->receive_timeout.tv_nsec =
611  (CE_DRIVER_FAF_RECEIVE_TIMEOUT_MS % 1000) * 1000000;
612  }
613 
614  /* Once configured/fixed, initialize this driver fsm */
615  if (fsm_init(&ce->driver.fsm, ce_driver_fsm_driver))
616  {
617  _dbg("Cannot initialize this command engine\n");
618  return 1;
619  }
620 
621  /* Assign default name, if not previously set */
622  if (ce->name == NULL)
623  {
624  ce->name = "YCP_NAME";
625  }
626 
627  /* Assign the external driver-fsm API */
628  ce->driver.send_command = send_command;
629  ce->driver.wait_command = wait_command;
630 
631  /* Basics: enable + auto-restart + start as soon as possible */
632  fsm_enable(&ce->driver.fsm);
633  fsm_set_loop(&ce->driver.fsm);
634  fsm_request_start(&ce->driver.fsm);
635 
636  /* fsm cycles */
637  fsm_print_info(&ce->driver.fsm);
638 
639  /* Let's go! */
640  return 0;
641 
642  /* Free _dbg() config */
643  #undef YCP_FNAME
644 }
