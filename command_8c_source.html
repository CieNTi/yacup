27 /* Locate a command in a set by its unique identifier
28  * Read `yacup/ce/command.h` for complete information. */
29 struct ce_command *ce_command_locate_by_id(struct ce_command_set *cmd_set,
30  size_t id)
31 {
32  /* Configure _dbg() */
33  #define YCP_FNAME "ce_command_locate_by_id"
34 
35  /* subset index */
36  size_t sx = 0;
37  /* command index */
38  size_t cx = 0;
39 
40  _dbg("Locating command 0x%02lX on set '%s' (%p)\n",
41  id,
42  cmd_set->name,
43  (void *)cmd_set);
44 
45  /* Iterate through subsets */
46  for (sx = 0; cmd_set->subset[sx] != NULL; sx++)
47  {
48  _dbg("Subset %lu: '%s' (%p)\n",
49  sx,
50  cmd_set->subset[sx]->name,
51  (void *)cmd_set->subset[sx]);
52 
53  /* Check if command is implemented */
54  for (cx = 0;
55  cmd_set->subset[sx]->command[cx] != NULL;
56  cx++)
57  {
58  /* id found? */
59  if (cmd_set->subset[sx]->command[cx]->id != id)
60  {
61  /* No, next */
62  continue;
63  }
64  _dbg("Command '%s' (subset %lu, id %lu) found\n",
65  cmd_set->subset[sx]->command[cx]->name,
66  sx,
67  cx);
68  return cmd_set->subset[sx]->command[cx];
69  }
70  }
71 
72  _dbg("Command not found\n");
73  return NULL;
74 
75  /* Free _dbg() config */
76  #undef YCP_FNAME
77 }

79 /* Validates a `ce_command`
80  * Read `yacup/ce/command.h` for complete information. */
81 struct ce_command *ce_command_validate(struct ce_command_set *cmd_set,
82  size_t id,
83  struct ce_command_argument *argument[])
84 {
85  /* Configure _dbg() */
86  #define YCP_FNAME "ce_command_validate"
87 
88  /* argument index */
89  size_t ax = 0;
90 
91  _dbg("Validating command 0x%02lX on set '%s' (%p)\n",
92  id, cmd_set->name, (void *)cmd_set);
93 
94  struct ce_command *ref_cmd = ce_command_locate_by_id(cmd_set, id);
95  if (ref_cmd == NULL)
96  {
97  _dbg("Command not found\n");
98  return NULL;
99  }
100 
101  _dbg("Command '%s' (0x%02lX) found, checking arguments\n",
102  ref_cmd->name, ref_cmd->id);
103 
104  /* No arguments passed, check if it fits */
105  if (argument == NULL)
106  {
107  if (ref_cmd->signature[0] != CE_DATA_NULL)
108  {
109  _dbg("Command expects at least 1 argument, none passed\n");
110 
111  /* Argument mismatch */
112  return NULL;
113  }
114 
115  /* Command found */
116  _dbg("Command '%s' (0x%02lX), without arguments, is valid\n",
117  ref_cmd->name, ref_cmd->id);
118  return ref_cmd;
119  }
120 
121  /* Check if command signature is valid */
122  for (ax = 0;
123  ((argument[ax] != NULL) &&
124  (ref_cmd->signature[ax] != CE_DATA_NULL));
125  ax++)
126  {
127  if (ref_cmd->signature[ax] !=
128  argument[ax]->type)
129  {
130  _dbg("Argument type mismatch %lu (%u vs. %u)\n",
131  ax, argument[ax]->type, ref_cmd->signature[ax]);
132 
133  /* Argument type mismatch */
134  return NULL;
135  }
136  }
137 
138  /* After the last command, both need to be NULL or one has extra arg */
139  if ((ref_cmd->signature[ax] != CE_DATA_NULL) ||
140  (argument[ax] != NULL))
141  {
142  _dbg("Extra argument not matched %lu (%u vs. %u)\n",
143  ax,
144  (argument[ax] == NULL)?CE_DATA_NULL:argument[ax]->type,
145  ref_cmd->signature[ax]);
146 
147  /* Extra argument not matched, next command */
148  return NULL;
149  }
150 
151  /* Command found */
152  _dbg("Command '%s' (0x%02lX) is valid\n", ref_cmd->name, ref_cmd->id);
153  return ref_cmd;
154 
155  /* Free _dbg() config */
156  #undef YCP_FNAME
157 }

159 /* Set a command listener, if found in a set
160  * Read `yacup/ce/command.h` for complete information. */
161 int ce_command_set_listener(struct ce_command_set *cmd_set,
162  size_t id,
163  struct ce_command_listener *listener)
164 {
165  /* Configure _dbg() */
166  #define YCP_FNAME "ce_command_set_listener"
167 
168  if (/* Invalid command set? */
169  (cmd_set == NULL) ||
170  /* Invalid listener structure? */
171  (listener == NULL)
172  /* Removed listener->listener check as we maybe just want the data
173  * and not the full callback+data. To be checked where executed */)
174  {
175  /* Cannot send, error */
176  _dbg("Invalid data when attaching a listener to a command\n");
177  return 1;
178  }
179 
180  struct ce_command *ref_cmd = NULL;
181  ref_cmd = ce_command_validate(cmd_set, id, listener->argument);
182 
183  /* Invalid command ? */
184  if (ref_cmd == NULL)
185  {
186  /* Cannot send, error */
187  _dbg("Cannot attach, unknown or invalid command\n");
188  return 1;
189  }
190 
191  /* Attach! */
192  ref_cmd->listener = listener;
193 
194  /* Ok! */
195  _dbg("Listener attached successfully\n");
196  return 0;
197 
198  /* Free _dbg() config */
199  #undef YCP_FNAME
200 }
