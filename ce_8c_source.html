27 /* Initializes a command engine referenced by a `ce` pointer
28  * Read `yacup/ce.h` for complete information. */
29 int ce_init(struct ce *ce,
30  int (*ce_driver_init)(struct ce *),
31  int (*ce_codec_driver_init)(struct ce_codec *),
32  int (*rb_driver_init)(struct rb *))
33 {
34  /* Configure _dbg() */
35  #define YCP_FNAME "ce_init"
36 
37  if (/* Invalid ce? */
38  (ce == NULL) ||
39  /* Invalid driver */
40  (ce_driver_init == NULL) ||
41  /* Defined codec driver? */
42  (ce_codec_driver_init == NULL) ||
43  /* Defined buffer driver? */
44  (rb_driver_init == NULL) ||
45  /* Invalid channels init */
46  ce_channel_init(&ce->out, ce_codec_driver_init, rb_driver_init) ||
47  ce_channel_init(&ce->in, ce_codec_driver_init, rb_driver_init) ||
48  /* And try to initialize this ce driver */
49  ce_driver_init(ce))
50  {
51  _dbg("Invalid ce or driver init function\n");
52  return 1;
53  }
54 
55  /* Assign default name, if not previously set */
56  if (ce->name == NULL)
57  {
58  ce->name = YCP_NAME;
59  }
60 
61  /* And finish! */
62  _dbg("ce '%s' initialized successfully\n", ce->name);
63  return 0;
64 
65  /* Free _dbg() config */
66  #undef YCP_FNAME
67 }

69 /* Validates command if found, encode data into a message and send it
70  * Read `yacup/ce.h` for complete information. */
71 int ce_send_command(struct ce *ce,
72  size_t id,
73  struct ce_command_argument *argument[])
74 {
75  /* Configure _dbg() */
76  #define YCP_FNAME "ce_send_command"
77 
78  /* Pointer to validated command */
79  struct ce_command *cmd_to_send = NULL;
80 
81  if (/* Invalid ce? */
82  (ce == NULL) ||
83  /* Invalid command set? */
84  (ce->out.command_set == NULL) ||
85  /* Invalid send_command() ? */
86  (ce->driver.send_command == NULL))
87  {
88  /* Cannot send, error */
89  _dbg("Error when sending a command: Invalid ce, command set or driver\n");
90  return 1;
91  }
92 
93  /* Look for it, validate and return a pointer if found and valid */
94  cmd_to_send = ce_command_validate(ce->out.command_set, id, argument);
95 
96  if (/* Invalid command */
97  (cmd_to_send == NULL) ||
98  /* Cannot send the command ? */
99  ce->driver.send_command(ce, cmd_to_send, argument))
100  {
101  /* Cannot send, error */
102  _dbg("Error when sending the command\n");
103  return 1;
104  }
105 
106  /* Ok! */
107  _dbg("Command sent\n");
108  return 0;
109 
110  /* Free _dbg() config */
111  #undef YCP_FNAME
112 }

114 /* Wait for a command and returns its id and data if success
115  * Read `yacup/ce.h` for complete information. */
116 int ce_wait_command(struct ce *ce,
117  size_t *id,
118  struct ce_command_argument **argument[])
119 {
120  /* Configure _dbg() */
121  #define YCP_FNAME "ce_wait_command"
122 
123  /* Pointer to validated command */
124  struct ce_command *cmd_to_wait = NULL;
125 
126  if (/* Invalid ce? */
127  (ce == NULL) ||
128  /* Invalid command ID holder? */
129  (id == NULL) ||
130  /* Invalid arguments ? */
131  (argument == NULL) ||
132  /* Invalid send_command() ? */
133  (ce->driver.wait_command == NULL))
134  {
135  /* Cannot send, error */
136  _dbg("Error when waiting commands: Invalid ce, id or argument holders\n");
137  return 1;
138  }
139 
140  /* Cannot receive any command ? */
141  if (ce->driver.wait_command(ce, id, argument))
142  {
143  /* Cannot send, error */
144  _dbg("Error when receiving the command\n");
145  return 1;
146  }
147 
148  /* Look for it, validate and return a pointer if found and valid */
149  cmd_to_wait = ce_command_validate(ce->in.command_set, *id, *argument);
150  if (cmd_to_wait == NULL)
151  {
152  /* Cannot send, error */
153  _dbg("Cannot validate command\n");
154  return 1;
155  }
156 
157  /* set id and go */
158  *id = cmd_to_wait->id;
159 
160  /* Ok! */
161  _dbg("Command received\n");
162  return 0;
163 
164  /* Free _dbg() config */
165  #undef YCP_FNAME
166 }
